### 실습 1. 구구단

void main() {
for (int i = 2; i < 10; i++) {
for (int j = 1; j < 10; j++) {
print('$i X $j =  ${j * i}');
}
print('');
}
}

### 실습 2.  사각형

void main() {
int n = 10; // 정사각형 크기

for (int i = 0; i < n; i++) {
print('

*'*

n);
}
print("");
for (int i = 0; i < n; i++) {
if (i == 0  i == n - 1) {
print('

*'*

n);
} else {
print('

*' + ' '*

(n - 2) + '

*');
}
}
print("");
for (int i = 0; i < n; i++) {
print(' '*

i + '/');
}
print("");
for (int i = 0; i < n; i++) {
print(' ' * (n - i - 1) + '\');
}
print("");
for (int i = 0; i < n; i++) {
String line = "";
for (int j = 0; j < n; j++) {
if (i == j  j == n - i - 1) {
line += '*';
} else {
line += ' ';
}
}
print(line);
}
}
